   }

\DeclareCiteCommand{\citeii}
  {\usebibmacro{prenote}} 
  {\printnames[familyonly]{author}%
   \setunit{\addspace}%
   \ifnumgreater{\value{labelname}}{2}
     {\addspace et al.}
     {}%
   \printtext{(}%
   \printfield{year}%
   \printtext{)}
   }
  {\multicitedelim}
  {\usebibmacro{postnote}}
